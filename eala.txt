-- [DRAGGABLE HELPER]
local function makeDraggable(obj)
    local d, ds, sp
    obj.InputBegan:Connect(function(i) 
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then 
            d = true; ds = i.Position; sp = obj.Position 
        end 
    end)
    obj.InputEnded:Connect(function(i) 
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then d = false end 
    end)
    UserInputService.InputChanged:Connect(function(i) 
        if d and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then 
            local dl = i.Position - ds
            obj.Position = UDim2.new(sp.X.Scale, sp.X.Offset + dl.X, sp.Y.Scale, sp.Y.Offset + dl.Y) 
        end 
    end)
end

local SavedConfig = { Toggles = {}, Pinned = {} }
local function saveToFile()
    if writefile then
        pcall(function()
            SavedConfig.Pinned = {}
            for name, data in pairs(_G.Pins or {}) do
                local pos = data.Container.Position
                SavedConfig.Pinned[name] = {x = pos.X.Scale, xo = pos.X.Offset, y = pos.Y.Scale, yo = pos.Y.Offset}
            end
            writefile("VanderConfig.json", HttpService:JSONEncode(SavedConfig))
        end)
    end
end
local function loadFromFile()
    if readfile and isfile and isfile("VanderConfig.json") then
        pcall(function()
            local decoded = HttpService:JSONDecode(readfile("VanderConfig.json"))
            if decoded then SavedConfig = decoded end
        end)
    end
end

local function createPinnedButton(labelText, callback, initialValue, savedPos, isToggle)
    if _G.Pins[labelText] then return end
    local active = initialValue
    local Container = create("Frame", {
        Size = UDim2.new(0, 100, 0, 30), Position = savedPos or UDim2.new(0.8, 0, 0.5, 0),
        BackgroundColor3 = COLORS.MainBG, BackgroundTransparency = 0.3, Parent = SG, ZIndex = 20

    local Stroke = create("UIStroke", {Color = (not isToggle or active) and COLORS.Border or COLORS.TextInactive, Thickness = 1.5, Parent = Container})
    
    local Label = create("TextLabel", {Size = UDim2.new(1, 0, 1, 0), BackgroundTransparency = 1, Text = labelText, TextColor3 = Color3.new(1,1,1), Font = "GothamBold", TextSize = 8, Parent = Container})

    -- FIXED DRAGGING & CLICKING FOR PINNED BUTTONS
    local d, ds, sp, moved = false
    Btn.InputBegan:Connect(function(i) 
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then 
            d = true; ds = i.Position; sp = Container.Position; moved = false
        end 
    end)
    UserInputService.InputChanged:Connect(function(i) 
        if d and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then 
            local dl = i.Position - ds
            if dl.Magnitude > 5 then moved = true end
            Container.Position = UDim2.new(sp.X.Scale, sp.X.Offset + dl.X, sp.Y.Scale, sp.Y.Offset + dl.Y) 
        end 
    end)
    UserInputService.InputEnded:Connect(function(i) 
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then 
            task.wait() -- Tiny delay to ensure moved flag is processed
            d = false 
        end 
    end)

    Btn.MouseButton1Click:Connect(function()
        if moved then return end 
        if isToggle then
            active = not active
            TweenService:Create(Stroke, TweenInfo.new(0.2), {Color = active and COLORS.Border or COLORS.TextInactive}):Play()
            callback(active)
        else
            TweenService:Create(Container, TweenInfo.new(0.1, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, 0, true), {BackgroundColor3 = COLORS.Border}):Play()
            callback()
        end
    end)
    
    _G.Pins[labelText] = {Container = Container, Update = function(v) if isToggle then active = v; Stroke.Color = v and COLORS.Border or COLORS.TextInactive end end}
end

local function scanPlots() 
    local newCache = {}
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return end
    for _, plot in pairs(plots:GetChildren()) do 
        local pods = plot:FindFirstChild("AnimalPodiums")
        if pods then 
            for _, pod in pairs(pods:GetChildren()) do if pod:FindFirstChild("Base") then table.insert(newCache, {plot = plot.Name, slot = pod.Name, pos = pod:GetPivot().Position}) end end 
        end
    end 
    GrabMod.Cache = newCache
end
local function executeSteal(prompt)
    task.spawn(function()
        if fireproximityprompt then pcall(function() fireproximityprompt(prompt) end)
        else local old = prompt.HoldDuration; prompt.HoldDuration = 0; prompt:InputBegan(Enum.UserInputType.MouseButton1) prompt:InputEnded(Enum.UserInputType.MouseButton1) prompt.HoldDuration = old end
    end)
end

-- SAFE FOLLOW Logic
local SafeFollow = { Enabled = false, Locked = false, DISTANCE = 6, targetChar = nil }
local function getNearPlayer(hrp)
    local near, d = nil, math.huge
    for _, p in pairs(Players:GetPlayers()) do
        if p ~= lp and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
            local dist = (p.Character.HumanoidRootPart.Position - hrp.Position).Magnitude
            if dist < d then d = dist near = p.Character end
        end
    end
    return near
end

-- COMBINED SPEED & FOLLOW Logic
RunService.Heartbeat:Connect(function()
    if lp.Character and lp.Character:FindFirstChild("HumanoidRootPart") and lp.Character:FindFirstChild("Humanoid") then
        local hrp, hum = lp.Character.HumanoidRootPart, lp.Character.Humanoid
        local dir = hum.MoveDirection
        if SpeedMod.Enabled then
            if SafeFollow.Enabled and SafeFollow.Locked then
                if not SafeFollow.targetChar or not SafeFollow.targetChar.Parent then SafeFollow.targetChar = getNearPlayer(hrp) end
                if SafeFollow.targetChar and SafeFollow.targetChar:FindFirstChild("HumanoidRootPart") then
                    local tHrp = SafeFollow.targetChar.HumanoidRootPart; local diff = (tHrp.Position - hrp.Position)
                    if diff.Magnitude > SafeFollow.DISTANCE then dir = (diff * Vector3.new(1,0,1)).Unit else dir = Vector3.new(0,0,0) end
                end
            end
            if dir.Magnitude > 0 then local targetX = hum.WalkSpeed < 25 and SpeedMod.Steal or SpeedMod.Speed; hrp.AssemblyLinearVelocity = Vector3.new(dir.X * targetX, hrp.AssemblyLinearVelocity.Y, dir.Z * targetX) end
        end
        if GrabMod.Enabled then
            local found = false
            for _, animal in pairs(GrabMod.Cache) do 
                if (hrp.Position - animal.pos).Magnitude <= GrabMod.Radius then
                    local p = workspace.Plots:FindFirstChild(animal.plot) and workspace.Plots[animal.plot].AnimalPodiums:FindFirstChild(animal.slot)
                    local prompt = p and p.Base:FindFirstChild("Spawn") and p.Base.Spawn:FindFirstChild("PromptAttachment") and p.Base.Spawn.PromptAttachment:FindFirstChildOfClass("ProximityPrompt")
                    if prompt and (prompt.ActionText:lower():find("steal") or prompt.ObjectText:lower():find("steal")) then executeSteal(prompt) found = true; GrabStatus.Text = "🔥 STEALING INSTANTLY! 🔥" GrabStatus.TextColor3 = Color3.fromRGB(255, 50, 70) end
                end 
            end
            if not found then GrabStatus.Text = "SCANNING FOR STEALABLE ANIMALS..." GrabStatus.TextColor3 = Color3.fromRGB(100, 100, 100) end
        end
    end
end)

-- ANTI RAGDOLL Logic
local AntiRag = { Enabled = false, Speed = 16, char = nil, conn = nil, move = nil }
local function cleanupRag()
    if AntiRag.char and AntiRag.char:FindFirstChild("HumanoidRootPart") and AntiRag.char.HumanoidRootPart:FindFirstChild("RagdollAnchor") then AntiRag.char.HumanoidRootPart.RagdollAnchor:Destroy() end
    if AntiRag.move then AntiRag.move:Disconnect() AntiRag.move = nil end
end
local function setupRag(char)
    AntiRag.char = char; cleanupRag()
    if AntiRag.conn then AntiRag.conn:Disconnect() end
    local rag = ReplicatedStorage:WaitForChild("Packages", 5):WaitForChild("Ragdoll", 5):WaitForChild("Ragdoll", 5)
    if not rag then return end
    AntiRag.conn = rag.OnClientEvent:Connect(function(a1, a2)
        if not AntiRag.Enabled then return end
        local root = char:FindFirstChild("HumanoidRootPart"); local hum = char:FindFirstChild("Humanoid")
        if a1 == "Make" or a2 == "manualM" then
            hum:ChangeState(Enum.HumanoidStateType.Freefall); root.CanCollide = false; cleanupRag()
            local anchor = create("BodyPosition", {Name = "RagdollAnchor", MaxForce = Vector3.new(1e6, 1e6, 1e6), P = 5000, D = 1000, Position = root.Position, Parent = root})
            AntiRag.move = RunService.RenderStepped:Connect(function() if not AntiRag.Enabled or not anchor or not anchor.Parent then cleanupRag() return end local d = hum.MoveDirection; anchor.Position = root.Position + (d.Magnitude > 0 and d.Unit * AntiRag.Speed or Vector3.new(0,0,0)) end)
        elseif a1 == "Destroy" or a2 == "manualD" then hum:ChangeState(Enum.HumanoidStateType.GettingUp); root.CanCollide = true; cleanupRag() end
    end)
end
