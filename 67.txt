local function makeSlider(parent, y, labelText, min, max, default)
    local enabled = false
    local value = default
    local Label = Instance.new("TextLabel", parent)
    Label.Text = labelText .. ": " .. value
    Label.Size = UDim2.new(1, 0, 0, 18)
    Label.Position = UDim2.new(0, 0, 0, y)
    Label.BackgroundTransparency = 1
    Label.Font = Enum.Font.GothamBold
    Label.TextSize = 12
    Label.TextColor3 = Color3.fromRGB(200,200,210)
    Label.TextXAlignment = Enum.TextXAlignment.Center
    local Bar = Instance.new("Frame", parent)
    Bar.Size = UDim2.new(0, BAR_W, 0, 12)
    Bar.Position = UDim2.new(0, (PANEL_W - BAR_W)/2, 0, y + 20)
    Bar.BackgroundColor3 = Color3.fromRGB(28,28,32)
    Instance.new("UICorner", Bar)
    local BarStroke = Instance.new("UIStroke", Bar)
    BarStroke.Thickness = 2
    BarStroke.Color = Color3.fromRGB(221,37,192)
    local p0 = (value - min) / (max - min)
    if p0 < 0 then p0 = 0 elseif p0 > 1 then p0 = 1 end
    local Fill = Instance.new("Frame", Bar)
    Fill.Size = UDim2.new(p0, 0, 1, 0)
    Fill.BackgroundColor3 = Color3.fromRGB(221,37,192)
    Instance.new("UICorner", Fill)
    local Dot = Instance.new("Frame", Bar)
    Dot.Size = UDim2.new(0, 12, 0, 12)
    Dot.AnchorPoint = Vector2.new(0.5, 0.5)
    Dot.Position = UDim2.new(p0, 0, 0.5, 0)
    Dot.BackgroundColor3 = Color3.fromRGB(245,110,215)
    Dot.ZIndex = 5
    Instance.new("UICorner", Dot)
    local Toggle = Instance.new("TextButton", parent)
    Toggle.Size = UDim2.new(0, TOGGLE_W, 0, 20)
    Toggle.Position = UDim2.new(0, (PANEL_W - BAR_W)/2 + BAR_W + 4, 0, y + 14)
    Toggle.BackgroundColor3 = Color3.fromRGB(22,22,26)
    Toggle.Font = Enum.Font.GothamBold
    Toggle.TextSize = 11
    Toggle.Text = "OFF"
    Toggle.TextColor3 = Color3.fromRGB(235,235,240)
    Instance.new("UICorner", Toggle)
    local function setState(state)
        enabled = state and true or false
        Toggle.Text = enabled and "ON" or "OFF"
    end
    Toggle.MouseButton1Click:Connect(function() setState(not enabled) end)
    local dragging = false
    local function update(input)
        local x = input.Position.X
        local bx = Bar.AbsolutePosition.X
        local bw = Bar.AbsoluteSize.X
        if bw <= 0 then return end
        local p = math.clamp((x - bx) / bw, 0, 1)
        value = math.floor(min + (max - min) * p + 0.5)
        Fill.Size = UDim2.new(p, 0, 1, 0)
        Dot.Position = UDim2.new(p, 0, 0.5, 0)
        Label.Text = labelText .. ": " .. value
    end
    Bar.InputBegan:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            update(i)
        end
    end)
    UserInputService.InputChanged:Connect(function(i)
        if dragging and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then
            update(i)
        end
    end)
    UserInputService.InputEnded:Connect(function(i)
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
            dragging = false
        end
    end)
    local function getter() return enabled, value end
    return getter, setState
end

local infJumpCon = {}
local jumpForce = 50
local clampFallSpeed = 80
local originalUseJumpPower = nil
local originalJumpPower = nil

local function DisableInfJump()
    for _,c in pairs(infJumpCon) do pcall(function() c:Disconnect() end) end
    infJumpCon = {}
    if Humanoid and originalUseJumpPower ~= nil and originalJumpPower ~= nil then
        pcall(function()
            Humanoid.UseJumpPower = originalUseJumpPower
            Humanoid.JumpPower = originalJumpPower
        end)
    end
end

local function EnableInfJump()
    if originalUseJumpPower == nil and Humanoid then
        pcall(function()
            originalUseJumpPower = Humanoid.UseJumpPower
            originalJumpPower = Humanoid.JumpPower
        end)
    end
    DisableInfJump()
    table.insert(infJumpCon, RunService.Heartbeat:Connect(function()
        if not infJumpEnabled then return end
        local char = LocalPlayer.Character
        if not char then return end
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if hrp and hrp.Velocity.Y < -clampFallSpeed then
            hrp.Velocity = Vector3.new(hrp.Velocity.X, -clampFallSpeed, hrp.Velocity.Z)
        end
    end))
    table.insert(infJumpCon, UserInputService.JumpRequest:Connect(function()
        if not infJumpEnabled then return end
        local char = LocalPlayer.Character
        if not char then return end
        local hrp = char:FindFirstChild("HumanoidRootPart")
        if hrp then
            local currentVelocity = hrp.Velocity
            local moveDirection = Humanoid.MoveDirection
            local horizontalVelocity = Vector3.new(currentVelocity.X, 0, currentVelocity.Z)
            if moveDirection.Magnitude > 0 then
                local moveDirectionNormalized = moveDirection.Unit
                local newHorizontalVelocity = moveDirectionNormalized * horizontalVelocity.Magnitude
                hrp.Velocity = Vector3.new(newHorizontalVelocity.X, jumpForce, newHorizontalVelocity.Z)
            else
                hrp.Velocity = Vector3.new(currentVelocity.X, jumpForce, currentVelocity.Z)
            end
        end
    end))
end

local unwalkConns = {}
local unwalkCharConn = nil
local storedAnims = {walk = {}, run = {}}
local idleAnimId = nil
local idleTrack = nil

local function storeAndRemoveWalkRun(char)
    local animate = char and char:FindFirstChild("Animate")
    if not animate then return end
    local idleFolder = animate:FindFirstChild("idle")
    if idleFolder then
        local a = idleFolder:FindFirstChildOfClass("Animation")
        if a then idleAnimId = a.AnimationId end
    end
    for _, tag in ipairs({"walk","run"}) do
        local folder = animate:FindFirstChild(tag)
        if folder then
            local children = folder:GetChildren()
            for i = #children,1,-1 do
                local child = children[i]
                if child:IsA("Animation") then
                    table.insert(storedAnims[tag], child.AnimationId)
                    child:Destroy()
                end
            end
        end
    end
end

local function restoreAnimsToAnimate(char)
    local animate = char and char:FindFirstChild("Animate")
    if not animate then return end
    for _, tag in ipairs({"walk","run"}) do
        local folder = animate:FindFirstChild(tag)
        if not folder then
            folder = Instance.new("Folder")
            folder.Name = tag
            folder.Parent = animate
        end
        for _, animId in ipairs(storedAnims[tag]) do
            if type(animId) == "string" then
                local newAnim = Instance.new("Animation")
                newAnim.Name = "restored"
                newAnim.AnimationId = animId
                newAnim.Parent = folder
            end
        end
        storedAnims[tag] = {}
    end
end

local function stopAllPlaying(hum)
    for _, t in pairs(hum:GetPlayingAnimationTracks()) do
        pcall(function() t:Stop() end)
    end
    if idleTrack then pcall(function() idleTrack:Stop() end) idleTrack = nil end
end

local function ensureIdlePlaying(hum)
    if not idleAnimId then return end
    if idleTrack and idleTrack.IsPlaying then return end
    local animObj = Instance.new("Animation")
    animObj.AnimationId = idleAnimId
    local ok, track = pcall(function() return hum:LoadAnimation(animObj) end)
    if ok and track then pcall(function() track:Play() end) idleTrack = track end
    pcall(function() animObj:Destroy() end)
end

local function startUnwalk()
    if unwalkCharConn then return end
    if LocalPlayer.Character then storeAndRemoveWalkRun(LocalPlayer.Character) end
    unwalkCharConn = LocalPlayer.CharacterAdded:Connect(function(c)
        task.wait(0.5)
        storeAndRemoveWalkRun(c)
    end)
    table.insert(unwalkConns, RunService.RenderStepped:Connect(function()
        local char = LocalPlayer.Character
        if not char then return end
        local hum = char:FindFirstChildOfClass("Humanoid")
        if not hum then return end
        if hum.MoveDirection.Magnitude > 0.1 then
            stopAllPlaying(hum)
        else
            ensureIdlePlaying(hum)
        end
    end))
end

local function stopUnwalk()
    if unwalkCharConn then unwalkCharConn:Disconnect() unwalkCharConn = nil end
    for _,c in pairs(unwalkConns) do pcall(function() c:Disconnect() end) end
    unwalkConns = {}
    local char = LocalPlayer.Character
    if char then restoreAnimsToAnimate(char) end
    if idleTrack then pcall(function() idleTrack:Stop() end) idleTrack = nil end
    idleAnimId = nil
end

getgenv().BestPetESP = getgenv().BestPetESP or {active = false, loop = nil, currentESP = nil}
local function parseValue(text)
    text = tostring(text or ""):gsub("%s", "")
    local num, suffix = text:match("([%d%.]+)([KkMmBbTt]?)")
    if not num then return 0 end
    num = tonumber(num) or 0
    local multipliers = {K=1e3, M=1e6, B=1e9, T=1e12}
    local mult = multipliers[(suffix or ""):upper()] or 1
    return num * mult
end

local function startESP()
    if getgenv().BestPetESP.active then return end
    getgenv().BestPetESP.active = true
    getgenv().BestPetESP.loop = task.spawn(function()
        while getgenv().BestPetESP.active do
            local debris = Workspace:FindFirstChild("Debris")
            if not debris then task.wait(0.5) continue end
            local bestPet = {value = -1, part = nil, text = "", display = "", template = nil}
            for _, template in ipairs(debris:GetChildren()) do
                if template.Name == "FastOverheadTemplate" then
                    local surfaceGui = template:FindFirstChildOfClass("SurfaceGui")
                    if not surfaceGui then continue end
                    local genLabel = surfaceGui:FindFirstChild("Generation", true)
                    if not genLabel or not genLabel:IsA("TextLabel") then continue end
                    local text = genLabel.Text or ""
                    if text ~= "" and (text:find("/s") or text:find("K") or text:find("M") or text:find("B")) then
                        local val = parseValue(text)
                        if val > bestPet.value then
                            local targetPart = surfaceGui.Adornee
                            if targetPart and targetPart:IsA("BasePart") then
                                local displayName = surfaceGui:FindFirstChild("DisplayName", true)
                                bestPet = {
                                    part = targetPart,
                                    value = val,
                                    text = text,
                                    display = displayName and displayName.Text or "Pet",
                                    template = template
                                }
                            end
                        end
                    end
                end
            end
            if bestPet.part and bestPet.part.Parent then
                createESP(bestPet.part, bestPet.display, bestPet.text)
            end

local function createMedusaForPlayer(plr)
    if medusaCircles[plr] then return end
    local char = plr.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local humanoid = char:FindFirstChildOfClass("Humanoid")
    if not hrp or not humanoid then return end
    local circle = Instance.new("CylinderHandleAdornment")
    circle.Name = "MedusaCircle_" .. plr.UserId
    circle.Adornee = hrp
    circle.Radius = MEDUSA_RANGE
    circle.Height = 0.1
    circle.Color3 = Color3.fromRGB(180, 0, 255)
    circle.Transparency = 0.35
    circle.AlwaysOnTop = true
    circle.ZIndex = 5
    circle.Parent = CoreGui
    medusaCircles[plr] = {circle = circle, humanoid = humanoid, hrp = hrp}

local function removeMedusaForPlayer(plr)
    local entry = medusaCircles[plr]
    if not entry then return end
    pcall(function() if entry.circle then entry.circle:Destroy() end end)
    medusaCircles[plr] = nil
end

local function removeMedusaCircles()
    for plr, entry in pairs(medusaCircles) do
        pcall(function() if entry.circle then entry.circle:Destroy() end end)
    end
    medusaCircles = {}
end

local function updateMedusaCircles()
    for plr, entry in pairs(medusaCircles) do
        if not plr or not entry then
            medusaCircles[plr] = nil
        else
            local hrp = entry.hrp
            local humanoid = entry.humanoid
            if hrp and humanoid and hrp.Parent then
                local offset = -(humanoid.HipHeight + 1.8)
                entry.circle.CFrame = CFrame.new(0, offset, 0) * CFrame.Angles(math.rad(90), 0, 0)
            else
                removeMedusaForPlayer(plr)
            end
        end
    end
end

local function updateMedusaList()
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
            if not medusaCircles[plr] then
                createMedusaForPlayer(plr)
            end
        end
    end
    for plr, entry in pairs(medusaCircles) do
        if not plr.Character or not plr.Character.Parent then
            removeMedusaForPlayer(plr)
        end
    end
end

local function getNearestPlayer(maxRange)
    maxRange = maxRange or math.huge
    if not LocalPlayer.Character then return nil end
    local myPos = HRP and HRP.Position or Vector3.new(0,0,0)
    local nearest = nil
    local shortest = maxRange
    for _, pl in ipairs(Players:GetPlayers()) do
        if pl ~= LocalPlayer and pl.Character and pl.Character:FindFirstChild("HumanoidRootPart") then
            local hrp = pl.Character.HumanoidRootPart
            local dist = (hrp.Position - myPos).Magnitude
            if dist < shortest then
                shortest = dist
                nearest = pl
            end
        end
    end
    return nearest
end

            task.wait(0.1)

        end)
        table.insert(playerESP_connections, charAddedConn)

    table.insert(playerESP_connections, playerAddedConn)

        removePlayerESP(plr)
    end
    for _, conn in ipairs(playerESP_connections) do

        pcall(function() if objs.hitbox then objs.hitbox:Destroy() end end)
        pcall(function() if objs.nameGui then objs.nameGui:Destroy() end end)
        playerESP_objects[key] = nil

    for _, child in pairs(CoreGui:GetChildren()) do
        if child:IsA("BoxHandleAdornment") and tostring(child.Name):match("^PlayerESP_Hitbox_") then
            pcall(function() child:Destroy() end)
        end
        if child:IsA("BillboardGui") and tostring(child.Name):match("^PlayerESP_Name_") then
            pcall(function() child:Destroy() end)
        end
    end
end
