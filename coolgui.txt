	local row = Instance.new("Frame")

	row.Size = UDim2.new(1, 0, 0, 50)
	row.Position = UDim2.new(0, 0, 0, yOffset)

	rowCorner.CornerRadius = UDim.new(0, 10)

	local label = Instance.new("TextLabel")
	label.Name = "Label"
	label.Size = UDim2.new(1, -94, 1, 0)
	label.Position = UDim2.new(0, 12, 0, 0)
	label.BackgroundTransparency = 1
	label.Text = labelText
	label.Font = Enum.Font.Kalam
	label.TextSize = 18
	label.TextColor3 = Color3.fromRGB(255,227,238)
	label.TextXAlignment = Enum.TextXAlignment.Left
	label.Parent = row

	local toggleTrack = Instance.new("Frame")
	toggleTrack.Name = "Track"
	toggleTrack.Size = UDim2.new(0, 56, 0, 28)
	toggleTrack.Position = UDim2.new(1, -76, 0.5, -14)

	local knob = Instance.new("Frame")
	knob.Name = "Knob"
	knob.Size = UDim2.new(0, 24, 0, 24)
	knob.Position = UDim2.new(0, 4, 0.5, -12)
	knob.BackgroundColor3 = Color3.fromRGB(255,255,255)
	knob.BorderSizePixel = 0
	knob.Parent = toggleTrack
	local knobCorner = Instance.new("UICorner")
	knobCorner.Parent = knob
	knobCorner.CornerRadius = UDim.new(1, 0)

	local toggleBtn = Instance.new("TextButton")
	toggleBtn.Name = "ToggleButton"
	toggleBtn.Text = ""
	toggleBtn.Size = UDim2.new(1, 0, 1, 0)
	toggleBtn.BackgroundTransparency = 1
	toggleBtn.Parent = toggleTrack

	local state = initialState and true or false
	local function updateVisual()
		if state then
			toggleTrack.BackgroundColor3 = Color3.fromRGB(222,37,194)
			knob.Position = UDim2.new(1, -28, 0.5, -12)
		else
			toggleTrack.BackgroundColor3 = Color3.fromRGB(64,64,72)
			knob.Position = UDim2.new(0, 4, 0.5, -12)
		end
	end

	toggleBtn.MouseButton1Click:Connect(function()
		state = not state
		updateVisual()
	end)

	updateVisual()
	return row, function() return state end
end

minusBtn.MouseButton1Click:Connect(function()
	if content.Visible then
		content.Visible = false
		main.Size = UDim2.new(0, 260, 0, 44)
		minusBtn.Text = "+"
	else
		content.Visible = true
		main.Size = UDim2.new(0, 260, 0, 188)
		minusBtn.Text = "-"
	end
end)

do
	local dragging = false
	local dragInput, dragStart, startPos
	local function blockInput(name, state, input) return Enum.ContextActionResult.Sink end

	titleBar.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPos = main.Position
			ContextActionService:BindAction("BlockInput", blockInput, false, Enum.UserInputType.Touch, Enum.UserInputType.MouseButton1)
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
					ContextActionService:UnbindAction("BlockInput")
				end
			end)
		end
	end)

	titleBar.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			local delta = input.Position - dragStart
			main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
		end
	end)
end
