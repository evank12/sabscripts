-- [DRAGGABLE HELPER]
local function makeDraggable(obj, targetFrame)
    local d, ds, sp
    obj.InputBegan:Connect(function(i) 
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then 
            d = true; ds = i.Position; sp = targetFrame.Position
            i:Consume()
        end 
    end)
    obj.InputEnded:Connect(function(i) 
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then 
            d = false 
        end 
    end)
    UserInputService.InputChanged:Connect(function(i) 
        if d and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then 
            local dl = i.Position - ds
            targetFrame.Position = UDim2.new(sp.X.Scale, sp.X.Offset + dl.X, sp.Y.Scale, sp.Y.Offset + dl.Y) 
        end 
    end)
end

-- [INFINITE JUMP]
-- =========================================
local jumpEnabled = false
local jumpConnection = nil
local platform = nil
local busy = false
local characterAddedConn = nil

local BOOST_HEIGHT = 7
local BOOST_TIME = 0.12
local OFFSET = 4.2

local function getRoot()
    local char = lp.Character
    return char and char:FindFirstChild("HumanoidRootPart")
end

local function ensurePlatform()
    if platform then return end
    platform = Instance.new("Part")
    platform.Size = Vector3.new(6, 1, 6)
    platform.Anchored = true
    platform.CanCollide = false
    platform.Transparency = 0.3
    platform.Name = "InfJumpPlatform"
    platform.Parent = workspace
end

local function doBoost()
    if not jumpEnabled or busy then return end
    busy = true
    local root = getRoot()
    if not root then busy = false return end
    ensurePlatform()
    local baseCF = root.CFrame - Vector3.new(0, OFFSET, 0)
    platform.CFrame = baseCF
    platform.CanCollide = true
    root.AssemblyLinearVelocity = Vector3.new(
        root.AssemblyLinearVelocity.X,
        60,
        root.AssemblyLinearVelocity.Z
    )
    TweenService:Create(platform, TweenInfo.new(BOOST_TIME, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
        CFrame = baseCF + Vector3.new(0, BOOST_HEIGHT, 0)
    }):Play()
    task.delay(0.08, function()
        if platform then platform.CanCollide = false end
    end)
    task.delay(BOOST_TIME, function() busy = false end)
end

local function enableJump()
    if jumpEnabled then return end
    jumpEnabled = true
    jumpConnection = UserInputService.JumpRequest:Connect(doBoost)
    characterAddedConn = lp.CharacterAdded:Connect(function()
        if platform then platform:Destroy() platform = nil end
    end)
end

local function disableJump()
    if not jumpEnabled then return end
    jumpEnabled = false
    if jumpConnection then jumpConnection:Disconnect() jumpConnection = nil end
    if characterAddedConn then characterAddedConn:Disconnect() characterAddedConn = nil end
    if platform then platform:Destroy() platform = nil end
end

-- =========================================
-- [AUTO-GRAB]
-- =========================================
local autoGrabEnabled = false
local autoGrabLoop = nil

local function getPos(prompt)
    local p = prompt.Parent
    if p:IsA("BasePart") then return p.Position end
    if p:IsA("Model") then
        local prim = p.PrimaryPart or p:FindFirstChildWhichIsA("BasePart")
        return prim and prim.Position
    end
    if p:IsA("Attachment") then return p.WorldPosition end
    local part = p:FindFirstChildWhichIsA("BasePart", true)
    return part and part.Position
end

local function findNearestStealPrompt()
    local root = getRoot()
    if not root then return nil end
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return nil end
    local myPos = root.Position
    local nearest = nil
    local nearestDist = math.huge
    for _, plot in ipairs(plots:GetChildren()) do
        for _, obj in ipairs(plot:GetDescendants()) do
            if obj:IsA("ProximityPrompt") and obj.Enabled and obj.ActionText == "Steal" then
                local pos = getPos(obj)
                if pos then
                    local dist = (myPos - pos).Magnitude
                    if dist <= obj.MaxActivationDistance and dist < nearestDist then
                        nearest = obj
                        nearestDist = dist
                    end
                end
            end
        end
    end
    return nearest
end

local function firePrompt(prompt)
    if not prompt then return end
    task.spawn(function()
        pcall(function()
            fireproximityprompt(prompt, 10000)
            prompt:InputHoldBegin()
            task.wait(0.04)
            prompt:InputHoldEnd()
        end)
    end)
end

local function startAutoGrab()
    if autoGrabLoop then return end
    autoGrabLoop = task.spawn(function()
        while autoGrabEnabled do
            local hum = lp.Character and lp.Character:FindFirstChildOfClass("Humanoid")
            if hum and hum.Health > 0 and hum.WalkSpeed > 29 then
                local nearest = findNearestStealPrompt()
                if nearest then
                    task.wait(0.07)
                    firePrompt(nearest)
                end
            end
            task.wait(0.3)
        end
        autoGrabLoop = nil
    end)
end

local function stopAutoGrab()
    autoGrabEnabled = false
    if autoGrabLoop then
        task.cancel(autoGrabLoop)
        autoGrabLoop = nil
    end
end

-- [OPTIMIZER + X-RAY (NON-FREEZING, TOGGLES OFF PROPERLY)]
-- =========================================
local optXrayData = {
    enabled = false,
    waterWaveOriginal = nil,
    globalShadowsOriginal = nil,
    materials = {},
    decals = {},
    xrayParts = {},
    conn = nil,
    applying = false
}

local function applyOptimizerAndXray()
    if optXrayData.applying then return end
    optXrayData.applying = true
    task.spawn(function()
        optXrayData.waterWaveOriginal = workspace.Terrain.WaterWaveSize
        optXrayData.globalShadowsOriginal = Lighting.GlobalShadows

        workspace.Terrain.WaterWaveSize = 0
        Lighting.GlobalShadows = false

        local allDescendants = game:GetDescendants()
        for i, obj in ipairs(allDescendants) do
            if not optXrayData.enabled then break end
            pcall(function()
                if obj:IsA("BasePart") then
                    optXrayData.materials[obj] = obj.Material
                    obj.Material = Enum.Plastic
                elseif obj:IsA("Decal") then
                    optXrayData.decals[obj] = obj.Transparency
                    obj.Transparency = 1
                end
            end)
            if (obj:IsA("BasePart") or obj:IsA("MeshPart")) then
                local name = obj.Name:lower()
                local parentName = obj.Parent and obj.Parent.Name:lower() or ""
                if name:find("base") or parentName:find("base") then
                    optXrayData.xrayParts[obj] = obj.LocalTransparencyModifier
                    obj.LocalTransparencyModifier = 0.8
                end
            end
            if i % 100 == 0 then task.wait() end
        end

        if optXrayData.enabled then
            optXrayData.conn = workspace.DescendantAdded:Connect(function(obj)
                task.wait(0.1)
                if not optXrayData.enabled then return end
                pcall(function()
                    if obj:IsA("BasePart") then
                        optXrayData.materials[obj] = obj.Material
                        obj.Material = Enum.Plastic
                    elseif obj:IsA("Decal") then
                        optXrayData.decals[obj] = obj.Transparency
                        obj.Transparency = 1
                    end
                end)
                if (obj:IsA("BasePart") or obj:IsA("MeshPart")) then
                    local name = obj.Name:lower()
                    local parentName = obj.Parent and obj.Parent.Name:lower() or ""
                    if name:find("base") or parentName:find("base") then
                        optXrayData.xrayParts[obj] = obj.LocalTransparencyModifier
                        obj.LocalTransparencyModifier = 0.8
                    end
                end
            end)
        end
        optXrayData.applying = false
    end)
end

local function revertOptimizerAndXray()
    optXrayData.enabled = false
    optXrayData.applying = false

    if optXrayData.conn then
        optXrayData.conn:Disconnect()
        optXrayData.conn = nil
    end

    local materialsSnapshot = optXrayData.materials
    local decalsSnapshot = optXrayData.decals
    local xraySnapshot = optXrayData.xrayParts

    optXrayData.materials = {}
    optXrayData.decals = {}
    optXrayData.xrayParts = {}

    task.spawn(function()
        if optXrayData.waterWaveOriginal then
            workspace.Terrain.WaterWaveSize = optXrayData.waterWaveOriginal
        end
        if optXrayData.globalShadowsOriginal ~= nil then
            Lighting.GlobalShadows = optXrayData.globalShadowsOriginal
        end

        local count = 0
        for obj, original in pairs(materialsSnapshot) do
            pcall(function() obj.Material = original end)
            count = count + 1
            if count % 100 == 0 then task.wait() end
        end

        count = 0
        for obj, original in pairs(decalsSnapshot) do
            pcall(function() obj.Transparency = original end)
            count = count + 1
            if count % 100 == 0 then task.wait() end
        end

        count = 0
        for obj, original in pairs(xraySnapshot) do
            pcall(function() obj.LocalTransparencyModifier = original end)
            count = count + 1
            if count % 100 == 0 then task.wait() end
        end

        optXrayData.waterWaveOriginal = nil
        optXrayData.globalShadowsOriginal = nil
    end)
end

-- [HITBOX VISUALS]
-- =========================================
local hitboxEnabled = false
local hitboxConnection = nil
local originalHitboxData = {}

local function applyHitbox(part)
    if not part then return end
    if not originalHitboxData[part] then
        originalHitboxData[part] = {
            Size = part.Size,
            Transparency = part.Transparency,
            Material = part.Material,
            CanCollide = part.CanCollide
        }
    end
    part.Size = Vector3.new(12,12,12)
    part.Transparency = 0.4
    part.Material = Enum.Material.SmoothPlastic
    part.CanCollide = false
end

local function revertHitbox(part)
    local data = originalHitboxData[part]
    if data then
        part.Size = data.Size
        part.Transparency = data.Transparency
        part.Material = data.Material
        part.CanCollide = data.CanCollide
    end
end

local function enableHitbox()
    hitboxEnabled = true
    hitboxConnection = RunService.RenderStepped:Connect(function()
        for _,p in pairs(Players:GetPlayers()) do
            if p ~= lp and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
                applyHitbox(p.Character.HumanoidRootPart)
            end
        end
    end)
end

local function disableHitbox()
    hitboxEnabled = false
    if hitboxConnection then
        hitboxConnection:Disconnect()
        hitboxConnection = nil
    end
    for part, _ in pairs(originalHitboxData) do
        revertHitbox(part)
    end
    originalHitboxData = {}
end

local function parseValue(text)
    text = tostring(text or ""):gsub("%s", "")
    local num, suffix = text:match("([%d%.]+)([KkMmBbTt]?)")
    if not num then return 0 end
    num = tonumber(num) or 0
    local multipliers = {K=1e3, M=1e6, B=1e9, T=1e12}
    local mult = multipliers[(suffix or ""):upper()] or 1
    return num * mult
end

local function enableBestESP()
    if bestESP.active then return end
    bestESP.active = true
    bestESP.loop = task.spawn(function()
        while bestESP.active do
            local debris = Workspace:FindFirstChild("Debris")
            if not debris then
                task.wait(0.5)
                continue
            end
            local bestPet = {value = -1, part = nil, text = "", display = "", template = nil}
            for _, template in ipairs(debris:GetChildren()) do
                if template.Name == "FastOverheadTemplate" then
                    local surfaceGui = template:FindFirstChildOfClass("SurfaceGui")
                    if not surfaceGui then continue end
                    local genLabel = surfaceGui:FindFirstChild("Generation", true)
                    if not genLabel or not genLabel:IsA("TextLabel") then continue end
                    local text = genLabel.Text or ""
                    if text ~= "" and (text:find("/s") or text:find("K") or text:find("M") or text:find("B")) then
                        local val = parseValue(text)
                        if val > bestPet.value then
                            local targetPart = surfaceGui.Adornee
                            if targetPart and targetPart:IsA("BasePart") then
                                local displayName = surfaceGui:FindFirstChild("DisplayName", true)
                                bestPet = {
                                    part = targetPart,
                                    value = val,
                                    text = text,
                                    display = displayName and displayName.Text or "Pet",
                                    template = template
                                }
                            end
                        end
                    end
                end
            end
            if bestPet.part and bestPet.part.Parent then
                createBestESP(bestPet.part, bestPet.display, bestPet.text)
            else
                if bestESP.currentESP then
                    pcall(function() bestESP.currentESP:Destroy() end)
                    bestESP.currentESP = nil
                end
            end
            task.wait(0.5)
        end
        if bestESP.currentESP then
            pcall(function() bestESP.currentESP:Destroy() end)
            bestESP.currentESP = nil
        end
    end)
end

        bestESP.loop = nil
    end
    if bestESP.currentESP then

        bestESP.currentESP = nil
    end
end

        timerESP.plotBillboards[plot] = nil
    end

local function initializeTimerESP()
    local plotsFolder = Workspace:FindFirstChild('Plots')
    if not plotsFolder then
        plotsFolder = Workspace:WaitForChild('Plots', 10)
        if not plotsFolder then return end
    end
    for _, plot in ipairs(plotsFolder:GetChildren()) do
        if plot:IsA('Model') then
            createTimerForPlot(plot)
        end
    end
    timerESP.childAddedConn = plotsFolder.ChildAdded:Connect(function(plot)
        if plot and plot:IsA('Model') then
            task.wait(0.2)
            createTimerForPlot(plot)
        end
    end)
    timerESP.childRemovedConn = plotsFolder.ChildRemoved:Connect(function(plot)
        removeTimerForPlot(plot)
    end)
end

local function updateTimerESP()
    for plot, gui in pairs(timerESP.plotBillboards) do
        if not plot or not gui or not gui.Parent then
            removeTimerForPlot(plot)
        else
            local ok, txt = pcall(function()
                if plot:FindFirstChild('Purchases') and plot.Purchases:FindFirstChild('PlotBlock') and plot.Purchases.PlotBlock:FindFirstChild('Main') then
                    local main = plot.Purchases.PlotBlock.Main
                    local billboard = main:FindFirstChild('BillboardGui')
                    if billboard and billboard:FindFirstChild('RemainingTime') and billboard.RemainingTime:IsA('TextLabel') then
                        return billboard.RemainingTime.Text
                    end
                end
                return nil
            end)
            if ok and txt and type(txt) == "string" and txt ~= "" then
                local label = gui:FindFirstChildWhichIsA('TextLabel', true) or gui:FindFirstChild('TextLabel')
                if label then
                    label.Text = (txt == '0s' and 'Unlocked') or txt
                    label.TextColor3 = (txt == '0s' and Color3.fromRGB(0,255,0)) or Color3.new(1,1,1)
                end
            else
                local label = gui:FindFirstChildWhichIsA('TextLabel', true) or gui:FindFirstChild('TextLabel')
                if label then label.Text = '' end
            end
        end
    end
end

local function muteBuzzing()
    if not antiEffectsEnabled then return end
    pcall(function()
        local playerScripts = lp:FindFirstChild("PlayerScripts")
        if playerScripts then
            local buzzing = playerScripts:FindFirstChild("Buzzing")
            if buzzing and buzzing:IsA("Sound") then
                pcall(function() buzzing:Destroy() end)
            end
        end
    end)
end

local function setupCameraHook()
    if metatableHooked then return end
    local camera = Workspace.CurrentCamera
    if not camera then return end

    local success, mt = pcall(function() return getrawmetatable(camera) end)
    if success and mt then
        local oldIndex = mt.__index
        local oldNewIndex = mt.__newindex
        if oldIndex and oldNewIndex then
            if setreadonly then pcall(setreadonly, mt, false) end
            pcall(function()
                mt.__newindex = function(self, key, value)
                    if antiEffectsEnabled and key == "FieldOfView" then
                        rawset(camera, "FieldOfView", 70)
                        return
                    end
                    return oldNewIndex(self, key, value)
                end
                mt.__index = function(self, key)
                    if antiEffectsEnabled and key == "FieldOfView" then
                        return 70
                    end
                    return oldIndex(self, key)
                end
            end)
            if setreadonly then pcall(setreadonly, mt, true) end
            metatableHooked = true
        end
    end
end

local function handleNewLightingChild(child)
    if not antiEffectsEnabled then return end
    pcall(function()
        if (child.Name == "BeeBlur" and child:IsA("BlurEffect")) or child:IsA("ColorCorrectionEffect") then
            child:Destroy()
        end
    end)
end

local function toggleAntiEffects(state)
    antiEffectsEnabled = state

    if antiEffectsEnabled then
        -- Turn ON
        removeControllers()
        setupCameraHook()
        
        table.insert(beeProtectionConnections, ReplicatedStorage.ChildAdded:Connect(function(child)
            if child and child.Name == "Controllers" then
                task.wait(0.25)
                removeControllers()
            end
        end))

        -- Heartbeat loop for FOV failsafe
        table.insert(beeProtectionConnections, RunService.Heartbeat:Connect(function()
            if Workspace.CurrentCamera and Workspace.CurrentCamera.FieldOfView ~= 70 then
                pcall(function() Workspace.CurrentCamera.FieldOfView = 70 end)
            end
        end))

        for _, effect in ipairs(Lighting:GetChildren()) do
            pcall(function()
                if effect.Name == "BeeBlur" or effect:IsA("ColorCorrectionEffect") then
                    effect:Destroy()
                end
            end)
        end

        table.insert(beeProtectionConnections, Lighting.ChildAdded:Connect(handleNewLightingChild))

        muteBuzzing()
        table.insert(beeProtectionConnections, lp.CharacterAdded:Connect(function()
            task.wait(0.3)
            muteBuzzing()
        end))
        
        print("Anti Bee/Disco Enabled")
    else
        -- Turn OFF
        for _, conn in ipairs(beeProtectionConnections) do
            if conn.Connected then
                conn:Disconnect()
            end
        end
        beeProtectionConnections = {}
        
        -- Reset FOV to default instantly when toggled off
        if Workspace.CurrentCamera then
            pcall(function() Workspace.CurrentCamera.FieldOfView = 70 end)
        end
        
        print("Anti Bee/Disco Disabled (Note: Destroyed controllers won't return until you respawn/reload)")
    end
end

    cleanupBoosterPhysics()

    lvAttach = Instance.new("Attachment")

    speedLV = Instance.new("LinearVelocity")
    speedLV.Name = "SpeedLV"
    speedLV.Attachment0 = lvAttach
    speedLV.VelocityConstraintMode = Enum.VelocityConstraintMode.Plane
    speedLV.PrimaryTangentAxis = Vector3.new(1, 0, 0)
    speedLV.SecondaryTangentAxis = Vector3.new(0, 0, 1)

    speedLV.Enabled = false
    speedLV.Parent = root

    jumpLV.Name = "JumpLV"
    jumpLV.Attachment0 = lvAttach
    jumpLV.VelocityConstraintMode = Enum.VelocityConstraintMode.Line
    jumpLV.LineDirection = Vector3.new(0, 1, 0)
    jumpLV.LineVelocity = 0
    jumpLV.MaxForce = 0
    jumpLV.RelativeTo = Enum.ActuatorRelativeTo.World
    jumpLV.Enabled = true
    jumpLV.Parent = root

    local hum = char:FindFirstChildOfClass("Humanoid")
    if hum then
        hum.Jumping:Connect(function(isJumping)
            if isJumping and boosterEnabled and jumpData.enabled and jumpData.value > 0 then
                jumpLV.LineVelocity = jumpData.value
                jumpLV.MaxForce = 1e6
                task.delay(0.15, function()
                    jumpLV.MaxForce = 0
                    jumpLV.LineVelocity = 0
                end)
            end
        end)
    end
end

    end
end)

    end)
    titleBar.InputEnded:Connect(function(i) 
        if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then d = false end 
    end)
    UserInputService.InputChanged:Connect(function(i) 
        if d and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then 
            local dl = i.Position - ds
            targetFrame.Position = UDim2.new(sp.X.Scale, sp.X.Offset + dl.X, sp.Y.Scale, sp.Y.Offset + dl.Y) 
        end 
    end)
end

        trackCorner.Parent = track

        local sliderKnob = Instance.new("Frame")
        sliderKnob.AnchorPoint = Vector2.new(0.5, 0.5)
        sliderKnob.BackgroundColor3 = COLORS.Border
        sliderKnob.BorderSizePixel = 0
        sliderKnob.Size = UDim2.new(0, 20, 0, 20)
        sliderKnob.Parent = track

        local knobCorner = Instance.new("UICorner")
        knobCorner.CornerRadius = UDim.new(1, 0)
        knobCorner.Parent = sliderKnob

        local counter = Instance.new("TextLabel")
        counter.Font = Enum.Font.GothamBold
        counter.Text = tostring(dataStore.value)
        counter.TextColor3 = COLORS.Border
        counter.TextScaled = true
        counter.BackgroundTransparency = 1
        counter.Position = UDim2.new(0.1, 0, 0.4, 0)
        counter.Size = UDim2.new(0, 50, 0, 25)
        counter.Parent = section

        local counterStroke = Instance.new("UIStroke")
        counterStroke.Color = COLORS.Border
        counterStroke.Thickness = 1
        counterStroke.Parent = counter

        local sliderDragging = false
        local lastInputId = nil

        local function updateFromInput(inputPos)
            local absPos = track.AbsolutePosition
            local absSize = track.AbsoluteSize
            local relX = (inputPos.X - absPos.X) / absSize.X
            relX = math.clamp(relX, 0, 1)
            local newVal = math.floor(minVal + (maxVal - minVal) * relX)
            dataStore.value = newVal
            counter.Text = tostring(newVal)
            sliderKnob.Position = UDim2.new(relX, 0, 0.5, 0)
        end

        track.InputBegan:Connect(function(input)
            if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) and not sliderDragging then
                updateFromInput(input.Position)
                sliderDragging = true
                lastInputId = input.UserInputType == Enum.UserInputType.Touch and input or nil
            end
        end)

        sliderKnob.InputBegan:Connect(function(input)
            if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) and not sliderDragging then
                sliderDragging = true
                lastInputId = input.UserInputType == Enum.UserInputType.Touch and input or nil
            end
        end)

        UserInputService.InputEnded:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                if lastInputId and input == lastInputId then
                    sliderDragging = false
                    lastInputId = nil
                elseif not lastInputId then
                    sliderDragging = false
                end
            end
        end)

        UserInputService.InputChanged:Connect(function(input)
            if sliderDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
                if lastInputId and input == lastInputId then
                    updateFromInput(input.Position)
                elseif not lastInputId and input.UserInputType == Enum.UserInputType.MouseMovement then
                    updateFromInput(input.Position)
                end
            end
        end)

        toggleBtn.MouseButton1Click:Connect(function()
            dataStore.enabled = not dataStore.enabled
            TweenService:Create(sliderDot, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
                Position = UDim2.new(dataStore.enabled and 0.75 or 0.25, 0, 0.5, 0)
            }):Play()
        end)

        local initialRel = (dataStore.value - minVal) / (maxVal - minVal)
        sliderKnob.Position = UDim2.new(initialRel, 0, 0.5, 0)
    end

    makeSliderSection("Speed", 0, 150, 55, speedData)
    makeSliderSection("Jump", 0, 150, 55, jumpData)
    makeSliderSection("Gravity", 0, 225, 50, gravData)

    boosterGui.Enabled = false
end
